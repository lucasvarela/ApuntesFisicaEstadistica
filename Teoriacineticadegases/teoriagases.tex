\chapter{Teoría Cinética de Gases}


\section{Distribución de Velocidades}


\begin{equation*}
f(\vec{r},\vec{\nu}) d\vec{r} d\vec{\nu} = \textbf{Número de moleculas que se encuentran en el intervalo $[\vec{r},\vec{r}+d\vec{r}]$ con velocidad en $[\vec{\nu},\vec{\nu}+d\vec{\nu}]$}
\end{equation*}

Se tiene un gas en un volumen $V$ y a temperatura $T$. Este gas contiene $N$ partículas, cada una con masa $m$. Se desea conocer propiedades del conjunto las partículas, como la distribución de velocidades para ellas. Para ello se usa el formalismo
canónico.

\begin{equation}
\label{1.1} \mathcal{H} = \sum_{i=1}^{N} \frac{|\vec{P}_i|^2}{2m} + \underset{1 \leq i < j \leq N}{\sum \sum} \Phi(|\vec{r}_i - \vec{r}_j|)
\end{equation}


El hamiltoniano para esta situación esta dado por la expresión (\ref{1.1}), donde el potencial $\Phi(|\vec{r}_i - \vec{r}_j|)$ solo depende de las posiciones relativas entre partículas.

\begin{equation}
\label{1.2}f(\vec{r},\vec{\nu}) = \text{cte } \int e^{-\beta \mathcal{H}(\vec{r},\vec{r}_2,\dots,\vec{r}_N,m\vec{\nu},m\vec{\nu}_2,\dots,m\vec{\nu}_N)} d\vec{r}_2\dots d\vec{r}_N d\vec{p}_2\dots d\vec{p}_N
\end{equation}

Se desea encontrar $f(\vec{r},\vec{\nu})$, que solo depende de la posición de una partícula y su velocidad(momento). Por lo que en la expresión (\ref{1.2}) se integra sobre las demás posiciones y momentos.
\begin{equation}
\label{1.3}f(\vec{r},\vec{\nu}) = \text{cte } e^{-\beta m |\vec{\nu}|^2 /2}\left(\int_{\mathbb{R}^3} d\vec{p}\; e^{-\beta |\vec{p}|^2/(2m)}\right)^{N-1} \int d\vec{r}_2\dots d\vec{r}_N \;e^{-\beta\underset{1 \leq i < j \leq N}{\sum \sum} \Phi(|\vec{r}_i - \vec{r}_j|) }
\end{equation}


La parte del momento se separa fácilmente y se integra sobre todos los momentos que no interesan. Para ver que la parte espacial no contribuye mas que con una constante, se procede a manipular las integrales de posición.


\begin{equation}
\label{1.4} \int d\vec{r}_2\dots d\vec{r}_N \;e^{-\beta\underset{1 \leq i < j \leq N}{\sum \sum} \Phi(|\vec{r}_i - \vec{r}_j|) } = \int d\vec{r}_2\dots d\vec{r}_N \;e^{-\beta\left( \sum_{j=2}^{N}\Phi(|\vec{r} - \vec{r}_j|) + \underset{2 \leq i < j \leq N}{\sum \sum} \Phi(|\vec{r}_i - \vec{r}_j|)\right) } 
\end{equation}



Haciendo un cambio de variable $ \vec{x}_j = \vec{r}_j - \vec{r}$

\begin{equation}
\label{1.5} \int d\vec{r}_2\dots d\vec{r}_N \;e^{-\beta\underset{1 \leq i < j \leq N}{\sum \sum} \Phi(|\vec{r}_i - \vec{r}_j|) } = \int d\vec{x}_2\dots d\vec{x}_N \;e^{-\beta\left( \sum_{j=2}^{N}\Phi(|\vec{x}|) + \underset{2 \leq i < j \leq N}{\sum \sum} \Phi(|\vec{x}_i - \vec{x}_j|)\right) } 
\end{equation}



Se puede observar en la ecuación (\ref{1.5}) que esta integral no depende de la posición $\vec{r}$. Por lo que $f(\vec{r},\vec{\nu})$ tampoco depende de la posición de la partícula. Esto se puede ver desde el punto físico debido a que la densidad no cambia al trasladar la partícula por lo que la distribución no debe depender de la posición. 


\begin{equation}
\label{1.6}f(\vec{r},\vec{\nu}) = C\; e^{-\beta m |\vec{\nu}|^2 /2}
\end{equation}



\begin{equation}
\label{1.7}\int_{\mathbb{R}^3} \int_{V}d\vec{r}d\vec{\nu}  \; f(\vec{r},\vec{\nu}) =1 
\end{equation}



Con la expresión (\ref{1.6}) se procede a calcular la constante $C$. Esta se encuentra exigiendo que la distribución este normalizada (\ref{1.7}).



\begin{equation}
\label{1.8}\int_{\mathbb{R}^3} \int_{V}d\vec{r}d\vec{\nu}  \; f(\vec{r},\vec{\nu}) = C V \left(\sqrt{\frac{2\pi }{m \beta}}\right)^3
\end{equation}



\begin{equation}
\label{1.9} f(\vec{r},\vec{\nu}) = n  \left({\frac{m }{2\pi k_B T }}\right)^{\frac{3}{2}} e^{-\beta m |\vec{\nu}|^2 /2}\\[0.4cm]
\end{equation}



Ahora se desea conocer solo la distribución de rapideces, $F(\nu)$.


\begin{equation*}
F({\nu})  = \textbf{Número de moleculas que se encuentran en el intervalo $[\vec{r},\vec{r}+d\vec{r}]$ con rapidez en $[{\nu},{\nu}+d{\nu}]$}
\end{equation*}




\begin{equation}
\label{1.10} f(\vec{r},\vec{\nu}) d\vec{\nu} = f(\vec{r},\vec{\nu})   \nu^2\sin \theta \;d\theta d\phi d\nu
\end{equation}


Pasando a coordenadas esféricas se obtiene la expresión (\ref{1.10}).


\begin{equation}
\label{1.11} F({\nu}) d{\nu} = \left(\int d \Omega \;f(\vec{r},\vec{\nu})   \right)\nu^2 d\nu
\end{equation}


\begin{equation}
\label{1.12}  \left(\int d \Omega \;f(\vec{r},\vec{\nu})   \right)\nu^2 d\nu = 4\pi \nu^2 f(\vec{r},\vec{\nu})    d\nu  
\end{equation}

Con el resultado (\ref{1.12}) se obtiene la famosa distribución de rapideces de Maxwell, dada por la siguiente expresión: 

\begin{equation}
\label{1.13} F({\nu}) d{\nu} =  4\pi n  \left({\frac{m }{2\pi k_B T }}\right)^{\frac{3}{2}} \nu^2e^{-\beta m {\nu}^2 /2}
\end{equation}


Con los resultados (\ref{1.9}) y (\ref{1.13}) se procede a dar algunas cantidades relevantes que se calculan a partir de ambas distribuciones. 




\begin{equation}
\label{1.14}  \braket{\vec{\nu}} = 0 
\end{equation}

La velocidad promedio de las partículas es cero, lo cual se intuye físicamente ya que las asunciones para el gas no tienen alguna dirección privilegiada a diferencia, por ejemplo, de un gas en un campo gravitacional.

\begin{equation}
\label{1.15}  \braket{{\nu}_{\mu}^2}= \int_{\mathbb{R}} d\nu_{\mu} \;\nu_{\mu}^2 \left({\frac{m }{2\pi k_B T }}\right)^{\frac{1}{2}} e^{-\beta m \nu_{\mu}^2 /2}= \frac{ k_B T}{m} \qquad \qquad \mu = x, y z
\end{equation}

La velocidad cuadrática promedio es igual en cualquier dirección. Nuevamente debido a que no hay ninguna dirección privilegiada. Con este resultado se prueba la equipartición de la energía en un gas que cumple las asunciones iniciales.

\begin{equation}
\label{1.16}  \braket{{\nu}^2}= \int_{\mathbb{R}^+} d\nu\;\nu^4 \left({\frac{m }{2\pi k_B T }}\right)^{\frac{3}{2}} e^{-\beta m \nu^2 /2}= \frac{3 k_B T}{m}
\end{equation}




\begin{equation}
\label{1.17}  \braket{|\vec{\nu}|^2} -|\braket{\vec{\nu}}|^2=   \braket{|\vec{\nu}|^2}  =  \braket{\nu_x^2 + \nu_y^2 +\nu_z^2} = \frac{3 k_B T}{m}
\end{equation}

Se observa que la velocidad cuadrática media es igual a la varianza de la velocidad.

\begin{equation}
\label{1.18} \braket{\nu} =  \braket{|\vec{\nu}|} =  \sqrt{\frac{8k_B T}{\pi m}} 
\end{equation}


\begin{equation}
\label{1.19} \braket{\nu} \neq \sqrt{\braket{\nu^2} }
\end{equation}


En la ecuación (\ref{1.19}) se reúnen dos resultados que muestran que la raíz cuadrada de la velocidad cuadrática media no es igual a la rapidez media.

\section{Función de Partición de un Oscilador Armónico Clásico}



\begin{equation}
\label{1.2.1} \mathcal{H}_{1D} = \frac{P_x^2}{2m} + \frac{1}{2}m \omega^2 x^2   
\end{equation}




Se considera un oscilador armónico en 1 dimensión, que tiene un hamiltoniano dado por (\ref{1.2.1}).




\begin{equation}
\label{1.2.2} \mathbf{Z}_c = \frac{1}{h}\int_{\mathbb{R}^2} dP_x dx \; e^{-\beta\left(\frac{P_x^2}{2m} + \frac{1}{2}m w^2 x^2   \right)}
\end{equation}

\begin{equation*}
 = \frac{1}{h}\sqrt{\frac{2\pi m }{\beta}} \sqrt{\frac{2\pi  }{\beta m \omega^2}} = \frac{2 \pi}{\beta\omega h}
\end{equation*}
Calculando la función de partición se obtiene el siguiente resultado:

\begin{equation}
\label{1.2.3} \mathbf{Z}_c = \frac{ k_B T}{\omega \hbar}
\end{equation}


Se procede a calcular la energía libre de Helmholtz y la energía interna:




\begin{equation}
\label{1.2.4} \beta \mathcal{F}_c = -\ln\left(\frac{ k_B T}{\omega \hbar}
\right)\end{equation}

\begin{equation}
\label{1.2.5} \braket{\mathcal{H}} = -\frac{\partial\ln \mathbf{Z}_c}{\partial \beta} = k_B T
\end{equation}




\begin{equation}
\label{1.2.6} \braket{\frac{1}{2m} P_x^2} = \frac{1}{\mathbf{Z}_c} \int_{\mathbf{R}} e^{-\frac{1}{2}\beta m \omega^2 x^2} dx
  \int_{\mathbf{R}} \frac{1}{2m} P_x^2e^{-\frac{1}{2m}\beta P_x^2} dP_x = \frac{1}{2} k_B T 
\end{equation}

\begin{equation}
\label{1.2.7} \braket{\frac{1}{2}m \omega^2 x^2} = \frac{1}{\mathbf{Z}_c} \int_{\mathbf{R}} \frac{1}{2}m \omega^2 x^2 e^{-\frac{1}{2}\beta m \omega^2 x^2} dx
  \int_{\mathbf{R}} e^{-\frac{1}{2m}\beta P_x^2} dP_x = \frac{1}{2} k_B T
 \end{equation}




En las ecuaciones (\ref{1.2.6}) y (\ref{1.2.7}) se puede ver que cada grado de libertad que aparece en forma \textbf{cuadrática} en el hamiltoniano contribuye con $\frac{1}{2} k_B T $ a la energía promedio.


\begin{equation}
\label{1.2.8} \mathbf{Z}_c = \left(\frac{ k_B T}{\omega \hbar}\right)^{3N}
\end{equation}

Para el tratamiento de $N$ osciladores tridimensionales se tiene la función de partición (\ref{1.2.8}).

\section{Sólido Cristalino Clásico}



\begin{equation}
\label{2.2.1} \mathcal{H} = \sum_{i=1}^{N}\frac{|\vec{P}_i|^2}{2m} +\frac{1}{2}\sum\sum K_i (x_i - x_{i-1})^2
\end{equation}


\begin{equation}
\label{2.2.2} \mathcal{U} = \frac{1}{2}\sum\sum K_i (x_i - x_{i-1})^2
\end{equation}

Un modelo para un solido cristalino esta dado por $N$ osciladores armónicos. El potencial $\mathcal{U}$ de interacción armónico es una forma cuadrática.

\begin{equation}
\label{2.2.3} \mathcal{U} =  \left(
\begin{array}{ccccc}
x_1 & . & . &.  & x_N\\
\end{array}
\right)\mathbb{V} \left(
\begin{array}{c}
x_1\\
.\\
.\\
.\\
x_N\\
\end{array}
\right)
\end{equation}

Cualquier forma cuadrática se puede escribir en forma matricial como se muestra en la ecuación (\ref{2.2.3}), donde $\mathbb{V}$ es una matriz simétrica.




\begin{equation}
\label{2.2.4} \mathbf{X} = \left(
\begin{array}{c}
x_1\\
.\\
.\\
.\\
x_N\\
\end{array}
\right)  = \mathbb{B} \left(
\begin{array}{c}
\xi_1\\
.\\
.\\
.\\
\xi_N\\
\end{array}
\right) = \mathbb{B} \; \mathbf{\xi}\\[0.3cm]
\end{equation}



Haciendo un cambio de base apropiado (\ref{2.2.4}) se puede diagonalizar la forma cuadrática:


\begin{equation}
\label{2.2.5} \mathbf{X}^t \mathbb{V}  \mathbf{X} = \xi^t \left( \mathbb{B}^t\mathbb{V} \mathbb{B} \right) \xi = \frac{1}{2}\xi^t \left(\begin{array}{ccccc}
m\omega_1 &\dots & 0 &  \dots& 0\\
\vdots& \ddots& & 0&\vdots\\
0& &\ddots & & 0 \\
\vdots& 0& & \ddots&\vdots \\
0&\dots & 0 &\dots &m\omega_{3N}\\
\end{array}\right) \xi\\[0.3cm]
\end{equation}




\begin{equation}
\label{2.2.6} \mathbf{X}^t \mathbb{V}  \mathbf{X} = \frac{1}{2} m \sum_{i=1}^{3N} \omega_i^2 \xi_i^2
\end{equation}



En la forma (\ref{2.2.6}) se puede extraer la función de partición sin dificultad:


\begin{equation}
\label{2.2.7} \mathbf{Z}_c = \prod_{j=1}^{3N} \frac{k_B T}{\hbar  \; \omega_j}
\end{equation}


Nuevamente, como en el caso de una dimensión, cada grado de libertad que aparece de forma cuadrática aporta un factor $\frac{1}{2}k_B T$:

\begin{equation}
\label{2.2.8} \braket{\frac{1}{2} \omega_j \xi_j^2} = \frac{1}{2} k_B T
\end{equation}


\begin{equation}
\label{2.2.9} \braket{\mathcal{H}} = 3N \left(\braket{P}^2 + \braket{\xi^2}\right) = 3N k_B T
\end{equation}

\begin{equation}
\label{2.2.10} U = 3N k_B T
\end{equation}


Utilizando la expresión para la energía interna del gas (\ref{2.2.10}), se procede a calcular la capacidad calorífica $C_V$:


\begin{equation}
\label{2.2.11} C_V = \frac{\partial U}{\partial T} =3N k_B 
\end{equation}

\section{Oscilador Armónico Cuántico}





\begin{equation}
\label{2.3.1} \widehat{H} = \frac{\widehat{P}^2}{2m} + \frac{1}{2} m \omega^2 \widehat{x}^2 
\end{equation}


Ahora se considera un solo oscilador armónico cuántico en una dimensión. El hamiltoniano para este caso por la ecuación (\ref{2.3.1}).





\begin{equation}
\label{2.3.2} \mathbf{Z}_c = \sum_{n} e^{-\beta E_n}
\end{equation}

\begin{equation}
\label{2.3.3} E_n = \left(n + \frac{1}{2}\right)\hbar \omega
\end{equation}




\begin{figure}[H]
 \label{figos}	\centering
 	\includegraphics[scale = 0.17]{oscilador}
 	\caption{Niveles de Energía de un Oscilador Armónico Cuántico}
 	\label{oscilador}
 \end{figure}
 

La función de partición cuántica esta dada por (\ref{2.3.2}), donde $E_n$ son las energías propias del operador energía $\widehat{H}$.




\begin{equation}
\label{2.3.4} \mathbf{Z}_c = \sum_{n=0}^{\infty} e^{-\beta \hbar \omega \left(n + \frac{1}{2}\right)} = e^{-\beta \hbar \omega /2}\sum_{n=0}^{\infty} e^{-n\beta \hbar \omega  } =\frac{e^{-\beta \hbar \omega /2}}{1 -e^{-\beta \hbar \omega }}
\end{equation}



\begin{equation}
\label{2.3.5} \braket{\widehat{H}} =\frac{1}{\mathbf{Z}_c}\sum_{n=0}^{\infty} e^{-\beta E_n}  =-\frac{\partial \ln \mathbf{Z}_c}{\partial \beta}
\end{equation}


\begin{equation}
\label{2.3.6} \braket{\widehat{H}} = \frac{\hbar \omega}{2} + \hbar \omega\frac{e^{-\beta \hbar \omega }}{1 -e^{-\beta \hbar \omega }}
\end{equation}


Con la expresión (\ref{2.3.6}) se procede a examinar el comportamiento de la energía interna a bajas y altas temperaturas. 



\begin{equation}
\label{2.3.7} \braket{\widehat{H}} = \frac{\hbar \omega}{2} + \hbar \omega\frac{1 {-\beta \hbar \omega }+ O(({\beta \hbar \omega })^2)}{{\beta \hbar \omega } +O(({\beta \hbar \omega })^2)} = \frac{1}{\beta} - \frac{\hbar \omega}{2} + O({\beta \hbar \omega })
\end{equation}


\begin{equation}
\label{2.3.8} \braket{\widehat{H}}= k_B T - \frac{\hbar \omega}{2} + O({\beta \hbar \omega })
\end{equation}


Se observa que para temperatura altas ($T k_B \gg \hbar \omega$) se recupera el resultado clásico.


\begin{equation}
\label{2.3.9} \braket{\widehat{H}} = \frac{\hbar \omega}{2}
 +\hbar \omega e^{-\beta \hbar \omega}   + O(\hbar \omega e^{-2\beta \hbar \omega} )\end{equation}


\begin{equation}
\label{2.3.10} C_V =  -\frac{\hbar^2 \omega^2}{k_B T^2} e^{-\beta \hbar \omega} \quad
 \substack{\xrightarrow{\hspace*{1cm}}\\ T \longrightarrow 0} \quad 0  
\end{equation}



El comportamiento para bajas temperaturas ($T k_B \ll \hbar \omega$) esta dado por las ecuaciones (\ref{2.3.9}) para (\ref{2.3.10})
